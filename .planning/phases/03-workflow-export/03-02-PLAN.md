    exportWorkflow() implementation steps:
    1. const project = await this.projectService.findOne(projectId);  // throws 404 on miss
    2. if (!project.workflowJson) return emptyExport(projectId, project.name);
    3. const nodes = await this.nodeRepo.find({ project: { id: projectId }, deletedAt: null }, { orderBy: { createdAt: 'asc' } });
    4. if (nodes.length === 0) return emptyExport(projectId, project.name);
    5. Parse edges: const rawEdges = (project.workflowJson.edges ?? []) as Array<{ sourceNodeId: string; targetNodeId: string }>;
    6. const nodeIds = new Set(nodes.map(n => n.nodeId));
    7. Build dependency map: buildDependencyMap(nodes, rawEdges, nodeIds) → Map<nodeId, string[]> (incoming deps)
    8. Build nodeMap: Map<nodeId, NodeMetadataEntity> for can_execute lookups
    9. Run Kahn: { order, cycle } = kahnSort(nodeIds, dependencyMap)
    10. if (cycle) throw new HttpException({ error: 'Cyclic dependency detected', cycle }, HttpStatus.UNPROCESSABLE_ENTITY);
    11. Build exportNodes array — for each node:
        - can_execute = (dependencyMap.get(n.nodeId) ?? []).every(

    Helper functions (pure — declare at module level, outside the class):

    function emptyExport(projectId: string, projectName: string): WorkflowExportResponse
      Returns: { projectId, projectName, exported_at: new Date().toISOString(), total_nodes: 0, nodes: [], execution_order: [], executable_now: [] }

    function buildDependencyMap(nodes, edges, nodeIds): Map<string, string[]>
      - Initialize map: for each node, map.set(node.nodeId, [])
      - For each edge: skip if sourceNodeId or targetNodeId not in nodeIds (dangling edge)
      - Otherwise: map.get(targetNodeId)!.push(sourceNodeId)  (incoming: B depends on A for edge A→B)
      - Return map

    function kahnSort(nodeIds, dependencyMap): { order: string[]; cycle: string[] | null }
      - Initialize inDegree: Map<string, number> — for each id in nodeIds:
        inDegree.set(id, (dependencyMap.get(id) ?? []).filter(d => nodeIds.has(d)).length)
        CRITICAL: Initialize from ALL nodeIds (not just nodes in dependencyMap) → isolated nodes get inDegree 0
      - Initial queue: [...nodeIds].filter(id => inDegree.get(id) === 0)
      - BFS: while queue not empty, shift curr, push to order
        For each nodeId in nodeIds where dependencyMap.get(nodeId)?.includes(curr):
          decrement inDegree.get(nodeId), if reaches 0 add to queue
      - if order.length < nodeIds.size: cycle exists → return { order: [], cycle: findCyclePath(nodeIds, dependencyMap, order) }
      - return { order, cycle: null }

    function findCyclePath(nodeIds, dependencyMap, processedOrder): string[]
      DFS with three-color marking. Track path: string[].
      Initialize: color = Map of all nodeIds to 'white'.

      Inner function dfs(node: string): string[] | null
        color.set(node, 'gray'); path.push(node)
        for dep of (dependencyMap.get(node) ?? []):
          if !nodeIds.has(dep): continue
          if color.get(dep) === 'gray':
            const cycleStart = path.indexOf(dep)
            return [...path.slice(cycleStart), dep]  // starts and ends with dep
          if color.get(dep) === 'white':
            const result = dfs(dep)
            if result: return result
        path.pop(); color.set(node, 'black'); return null

      const remaining = [...nodeIds].filter(id => !processedOrder.includes(id))
      for start of remaining:
        if color.get(start) === 'white':
          path.length = 0
          const result = dfs(start)
          if result: return result
      return [remaining[0], remaining[1] ?? remaining[0], remaining[0]]  // fallback (unreachable)
