interface WorkflowExportResponse {
  projectId: string;
  projectName: string;
  exported_at: string;    // ISO 8601
  total_nodes: number;
  nodes: ExportNode[];

interface ExportNode {
  nodeId: string;
  type: string;
  requirement: string;
  prompt: string | null;
  attributes: Array<{ key: string; value: string }> | null;
  status: NodeStatus;
  dependencies: string[];   // incoming edge sourceNodeIds (in-scope only)
  can_execute: boolean;     // true iff all dependencies have status === 'completed' (or out of scope)
}
